# Pod with curl alpine
```
cat <<EOF | kubectl apply -f -
apiVersion: v1
kind: Pod
metadata:
  name: cam-curl-alpine
  namespace: default
  labels:
    cam: curl-alpine
  annotations:
    author: cameda
spec:
  containers:
  - name: curl-alpine
    image: alpine/curl
    imagePullPolicy: IfNotPresent
    resources:
      requests:
        cpu: "20m"
        memory: "20Mi"
      limits:
        memory: "120Mi"
    command: ["sh", "-c"]
    args: ["sleep 3650d"]
  restartPolicy: OnFailure
  hostname: curl
  nodeSelector:
    kubernetes.io/os: linux
EOF
