# Pod with traefik
```
cat <<EOF | kubectl apply -f -
apiVersion: v1
kind: Pod
metadata:
  name: cam-pod-traefik
  namespace: default
  labels:
    cam: traefik
  annotations:
    author: cameda
spec:
  containers:
  - name: traefik
    image: traefik
    imagePullPolicy: IfNotPresent
    resources:
      requests:
        cpu: "100m"
        memory: "100Mi"
      limits:
        memory: "120Mi"
  restartPolicy: OnFailure
  hostname: traefik
  nodeSelector:
    kubernetes.io/os: linux
EOF
